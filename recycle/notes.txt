setup_stack:
	cli										; clear interupts
	mov ax,RELOC_SEG
	add ax,BOOT_SECTOR_SIZE
	mov ss,ax								; Stack is in the same segment
	mov ax,STACK_SIZE						; Setting stack pointer
	mov sp,ax
	sti										; start interupts
	ret
